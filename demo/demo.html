Usage

		  /*
		   * If you just want to initialize the modal and bind/show it later
		   */
		  
		  reticle.new({
		    content: document.getElementById('some-element')
		  });
		  
		  /*
		   * If you want to go ahead and bind some button(s) to reticle.show
		   */
		  reticle.new({
		    buttonClass: "someClass",
		    content: document.getElementById('some-element')
		  })
